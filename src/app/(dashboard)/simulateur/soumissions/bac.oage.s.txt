"use client";

import { useEffect, useMemo, useState } from "react";
import Link from "next/link";
import { toast } from "sonner";
import { AttemptHeader } from "./components/AttemptHeader";
import { AttemptDetails } from "./components/AttemptDetails";

type Attempt = {
  id: string;
  combination_id: string;
  status: "in_progress" | "submitted" | "graded";
  started_at?: string | null;
  submitted_at?: string | null;
  duration_seconds?: number | null;
};

type Evaluation = {
  task_number: number | null;
  score_20?: number | null;
  cecr_level?: string | null;
  positives?: string[] | null;
  improvements?: string[] | null;
  suggested_correction?: string | null;
  created_at?: string;
};

export default function SubmissionsPage() {
  const [attempts, setAttempts] = useState<Attempt[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const [openId, setOpenId] = useState<string | null>(null);
  const [evals, setEvals] = useState<Record<string, Evaluation[]>>({});
  const [detailsLoading, setDetailsLoading] = useState<Record<string, boolean>>({});
  const [activeTabs, setActiveTabs] = useState<Record<string, 'notes' | 'feedback' | 'checklist' | 'c2'>>({});
  const [evalErrors, setEvalErrors] = useState<Record<string, string>>({});
  const [evalMeta, setEvalMeta] = useState<Record<string, { model_used?: string; inserted?: number }>>({});
  // Minimal OpenAI health info
  const [openaiHealth, setOpenaiHealth] = useState<{ model?: string; hasKey?: boolean } | null>(null);
  // Inline per-task OpenAI evaluation results
  const [aiTaskResults, setAiTaskResults] = useState<Record<string, Evaluation>>({});
  const [aiTaskLoading, setAiTaskLoading] = useState<Record<string, boolean>>({});
  const [detailsMap, setDetailsMap] = useState<Record<string, { tasks: any[]; answers: { task_number: number; content: string; word_count?: number }[] }>>({});

  // Tab helpers (per attemptId + evaluation index)
  const tabKey = (attemptId: string, idx: number) => `${attemptId}:${idx}`;
  const setTab = (attemptId: string, idx: number, tab: 'notes' | 'feedback' | 'checklist' | 'c2') => {
    setActiveTabs(prev => ({ ...prev, [tabKey(attemptId, idx)]: tab }));
  };

  // Sequential per-task evaluation (eco mode): runs task 1 -> 2 -> 3
  const runPerTask = async (id: string, dry: boolean) => {
    if (perTaskRunning[id]) return; // already running
    setPerTaskRunning(prev => ({ ...prev, [id]: { step: 0, dry } }));
    try {
      for (let task = 1 as 1|2|3; task <= 3; task = (task + 1) as 1|2|3) {
        setPerTaskRunning(prev => ({ ...prev, [id]: { step: task, dry } }));
        toast.info(`${dry ? 'Prévisualisation' : 'Évaluation'} tâche T${task}…`);
        setDetailsLoading(prev => ({ ...prev, [id]: true }));
        try {
          const url = `/api/simulator/attempts/${encodeURIComponent(id)}/evaluate?per_task=1&task=${task}${dry ? '&dry=1' : ''}`;
          const r = await fetch(url, { method: 'POST' });
          const j = await r.json().catch(() => ({}));
          if (!r.ok) {
            const msg = j?.error || `${dry ? 'Prévisualisation' : 'Évaluation'} T${task} échouée (HTTP ${r.status})`;
            setEvalErrors(prev => ({ ...prev, [id]: msg }));
            toast.error(msg);
            break; // stop sequence on error
          }
          // diagnostics
          setEvalMeta(prev => ({ ...prev, [id]: {
            model_used: j?.model_used,
            provider_status: j?.provider_status,
            payload_bytes: j?.payload_bytes,
            content_len: j?.content_len,
            inserted: j?.inserted,
          }}));
          if (dry) {
            const arr: Evaluation[] = Array.isArray(j?.preview?.evaluations) ? j.preview.evaluations : [];
            // merge preview with existing preview set
            setPreviewEvals(prev => {
              const ex = prev[id] || [];
              // replace or add this task
              const filtered = ex.filter(e => e.task_number !== task);
              return { ...prev, [id]: [...filtered, ...arr] };
            });
          } else {
            // non-dry: refetch saved evaluations to display immediately
            try {
              const r2 = await fetch(`/api/simulator/attempts/${encodeURIComponent(id)}`);
              const j2 = await r2.json();
              if (r2.ok) {
                const arr: Evaluation[] = Array.isArray(j2?.data?.evaluations) ? j2.data.evaluations : [];
                setEvals(prev => ({ ...prev, [id]: arr }));
              }
            } catch {}
          }
        } finally {
          setDetailsLoading(prev => ({ ...prev, [id]: false }));
        }
      }
      toast.success(`${dry ? 'Prévisualisation' : 'Évaluation'} par tâches terminée`);
    } finally {
      setPerTaskRunning(prev => {
        const { [id]: _, ...rest } = prev as any;
        return rest;
      });
    }
  };

  
  const getTab = (attemptId: string, idx: number) => activeTabs[tabKey(attemptId, idx)] || 'notes';

  useEffect(() => {
    (async () => {
      try {
        const res = await fetch('/api/simulator/attempts');
        if (res.status === 401) {
          setError('Vous devez être connecté pour voir vos soumissions.');
          return;
        }
        const json = await res.json();
        if (!res.ok) throw new Error(json?.error || 'Erreur chargement');
        const list = json?.data || [];
        setAttempts(list);
        // Auto-open the last submitted attempt (from localStorage) on first load
        try {
          const last = localStorage.getItem('last_submitted_attempt') || '';
          if (last && list.some((a: Attempt) => a.id === last)) {
            setOpenId(last);
          }
        } catch {}
      } catch (e: any) {
        setError(e?.message || 'Erreur inconnue');
      } finally {
        setLoading(false);
      }
    })();
  }, []);

  // Polling: refresh while there are submitted attempts
  useEffect(() => {
    if (loading) return;
    if (error) return; // stop polling if unauthorized or error
    const hasSubmitted = attempts.some(a => a.status === 'submitted');
    if (!hasSubmitted) return;
    const id = setInterval(async () => {
      try {
        const res = await fetch('/api/simulator/attempts');
        if (res.status === 401) { setError('Vous devez être connecté pour voir vos soumissions.'); clearInterval(id); return; }
        const json = await res.json();
        if (res.ok) {
          setAttempts(json?.data || []);
          if (openId) {
            // refresh open details as well
            const r2 = await fetch(`/api/simulator/attempts/${encodeURIComponent(openId)}`);
            if (r2.status === 401) { setError('Vous devez être connecté pour voir vos soumissions.'); clearInterval(id); return; }
            const j2 = await r2.json();
            if (r2.ok) {
              const arr: Evaluation[] = Array.isArray(j2?.data?.evaluations) ? j2.data.evaluations : [];
              setEvals(prev => ({ ...prev, [openId]: arr }));
            }
          }
        }
      } catch {}
    }, 6000);
    return () => clearInterval(id);
  }, [attempts, loading, openId, error]);

  // Focused polling for the open attempt: poll details every 3s until evaluations exist or status is graded
  useEffect(() => {
    if (!openId) return;
    const current = attempts.find(a => a.id === openId);
    if (!current) return;
    // If already graded and we have evals, stop
    if (current.status === 'graded' && Array.isArray(evals[openId]) && evals[openId].length > 0) return;
    let stop = false;
    const tick = async () => {
      try {
        const r = await fetch(`/api/simulator/attempts/${encodeURIComponent(openId)}`);
        if (r.status === 401) { setError('Vous devez être connecté pour voir vos soumissions.'); stop = true; return; }
        const j = await r.json();
        if (r.ok) {
          const arr: Evaluation[] = Array.isArray(j?.data?.evaluations) ? j.data.evaluations : [];
          if (arr.length > 0) {
            setEvals(prev => ({ ...prev, [openId]: arr }));
            // Clear hint in storage
            try { localStorage.removeItem('last_submitted_attempt'); } catch {}
            stop = true;
          }
        }
      } catch {}
    };

    // Immediate try then interval
    tick();
    const id = setInterval(() => { if (!stop) tick(); else clearInterval(id); }, 3000);
    return () => clearInterval(id);
  }, [openId, attempts]);

  // Manual retrigger of evaluation for a submitted attempt
  const retrigger = async (id: string) => {
    try {
      toast.info('Relance de l\'évaluation IA…');
      const r = await fetch(`/api/simulator/attempts/${encodeURIComponent(id)}/evaluate`, { method: 'POST' });
      const j = await r.json().catch(() => ({}));
      if (!r.ok) {
        const msg = j?.error || `Échec (HTTP ${r.status})`;
        setEvalErrors(prev => ({ ...prev, [id]: msg }));
        toast.error(`Échec de la relance: ${msg}`);
      } else {
        toast.success('Évaluation relancée');
        setEvalMeta(prev => ({ ...prev, [id]: {
          model_used: j?.model_used,
          provider_status: j?.provider_status,
          payload_bytes: j?.payload_bytes,
          content_len: j?.content_len,
          inserted: j?.inserted,
        }}));
        // If rows inserted, refresh details now to auto-display feedback
        if (typeof j?.inserted === 'number' && j.inserted > 0) {
          setOpenId(id);
          setDetailsLoading(prev => ({ ...prev, [id]: true }));
          try {
            const r2 = await fetch(`/api/simulator/attempts/${encodeURIComponent(id)}`);
            const j2 = await r2.json();
            if (r2.ok) {
              const arr: Evaluation[] = Array.isArray(j2?.data?.evaluations) ? j2.data.evaluations : [];
              setEvals(prev => ({ ...prev, [id]: arr }));
            }
          } catch (e: any) {
            console.error(e);
          } finally {
            setDetailsLoading(prev => ({ ...prev, [id]: false }));
          }
        }
      }
    } catch (e: any) {
      toast.error(e?.message || 'Erreur relance');
    }
  };

  const toggleOpen = async (id: string) => {
    if (openId === id) {
      setOpenId(null);
      return;
    }
    setOpenId(id);
    if (!evals[id]) {
      setDetailsLoading(prev => ({ ...prev, [id]: true }));
      try {
        const res = await fetch(`/api/simulator/attempts/${encodeURIComponent(id)}`);
        const json = await res.json();
        if (!res.ok) throw new Error(json?.error || 'Erreur détails');
        const arr: Evaluation[] = Array.isArray(json?.data?.evaluations) ? json.data.evaluations : [];
        setEvals(prev => ({ ...prev, [id]: arr }));
        const tasks = Array.isArray(json?.data?.tasks) ? json.data.tasks : [];
        const answers = Array.isArray(json?.data?.answers) ? json.data.answers : [];
        setDetailsMap(prev => ({ ...prev, [id]: { tasks, answers } }));
        const st = (json?.data?.attempt?.status || '') as string;
        if (st === 'submitted') {
          toast.info('Évaluation en cours, réessayez dans quelques secondes…');
          // Auto-trigger evaluation once if not yet triggered
          if (!autoTriggered[id]) {
            try {
              setAutoTriggered(prev => ({ ...prev, [id]: true }));
              const trig = await fetch(`/api/simulator/attempts/${encodeURIComponent(id)}/evaluate`, { method: 'POST' });
              const tj = await trig.json().catch(() => ({}));
              if (!trig.ok) {
                const msg = tj?.error || `Échec évaluation (HTTP ${trig.status})`;
                setEvalErrors(prev => ({ ...prev, [id]: msg }));
                console.warn('[soumissions] auto-trigger evaluate failed', trig.status, tj?.error);
              } else {
                setEvalMeta(prev => ({ ...prev, [id]: {
                  model_used: tj?.model_used,
                  provider_status: tj?.provider_status,
                  payload_bytes: tj?.payload_bytes,
                  content_len: tj?.content_len,
                  inserted: tj?.inserted,
                }}));
                if (typeof tj?.inserted === 'number' && tj.inserted > 0) {
                  // fetch details immediately
                  try {
                    const r2 = await fetch(`/api/simulator/attempts/${encodeURIComponent(id)}`);
                    const j2 = await r2.json();
                    if (r2.ok) {
                      const arr: Evaluation[] = Array.isArray(j2?.data?.evaluations) ? j2.data.evaluations : [];
                      setEvals(prev => ({ ...prev, [id]: arr }));
                    }
                  } catch {}
                }
              }
            } catch (e: any) {
              const msg = e?.message || 'Erreur déclenchement évaluation';
              setEvalErrors(prev => ({ ...prev, [id]: msg }));
              console.warn('[soumissions] auto-trigger evaluate err', msg);
            }
          }
          // Ping health endpoint to confirm env/model readiness
          try {
            setHealth({ loading: true });
            const h = await fetch('/api/simulator/health');
            const hj = await h.json();
            if (h.ok) {
              setHealth({
                loading: false,
                ok: Boolean(hj?.ok),
                model: hj?.openrouter?.model,
                baseUrl: hj?.openrouter?.baseUrl,
                hasKey: Boolean(hj?.openrouter?.hasKey),
                timestamp: hj?.timestamp,
              });
            } else {
              setHealth({ loading: false, ok: false, error: hj?.error || 'Health check failed' });
            }
          } catch (e: any) {
            setHealth({ loading: false, ok: false, error: e?.message || 'Health check error' });
          }
        } else {
          setHealth(null);
        }
      } catch (e: any) {
        toast.error(e?.message || 'Erreur chargement feedbacks');
      } finally {
        setDetailsLoading(prev => ({ ...prev, [id]: false }));
      }
    }
  };

  // Dry-run preview: get feedback without saving to DB
  const preview = async (id: string) => {
    try {
      toast.info('Prévisualisation du feedback (sans sauvegarde)…');
      setDetailsLoading(prev => ({ ...prev, [id]: true }));
      const r = await fetch(`/api/simulator/attempts/${encodeURIComponent(id)}/evaluate?dry=1`, { method: 'POST' });
      const j = await r.json().catch(() => ({}));
      if (!r.ok) {
        const msg = j?.error || `Prévisualisation échouée (HTTP ${r.status})`;
        setEvalErrors(prev => ({ ...prev, [id]: msg }));
        toast.error(msg);
        return;
      }
      const arr: Evaluation[] = Array.isArray(j?.preview?.evaluations) ? j.preview.evaluations : [];
      setPreviewEvals(prev => ({ ...prev, [id]: arr }));
      setEvalMeta(prev => ({ ...prev, [id]: {
        model_used: j?.model_used,
        provider_status: j?.provider_status,
        payload_bytes: j?.payload_bytes,
        content_len: j?.content_len,
        inserted: j?.inserted,
      }}));
      toast.success('Prévisualisation prête');
    } catch (e: any) {
      toast.error(e?.message || 'Erreur de prévisualisation');
    } finally {
      setDetailsLoading(prev => ({ ...prev, [id]: false }));
    }
  };

  // While an attempt is open and submitted, auto-refresh health every 10s
  useEffect(() => {
    if (!openId) return;
    const current = attempts.find(a => a.id === openId);
    if (!current || current.status !== 'submitted') return;
    let alive = true;
    const tick = async () => {
      try {
        const h = await fetch('/api/simulator/health');
        const hj = await h.json();
        if (!alive) return;
        if (h.ok) {
          setHealth({
            loading: false,
            ok: Boolean(hj?.ok),
            model: hj?.openrouter?.model,
            baseUrl: hj?.openrouter?.baseUrl,
            hasKey: Boolean(hj?.openrouter?.hasKey),
            timestamp: hj?.timestamp,
          });
        } else {
          setHealth({ loading: false, ok: false, error: hj?.error || 'Health check failed' });
        }
      } catch (e: any) {
        if (!alive) return;
        setHealth({ loading: false, ok: false, error: e?.message || 'Health check error' });
      }
    };
    tick();
    const id = setInterval(tick, 10000);
    return () => { alive = false; clearInterval(id); };
  }, [openId, attempts]);

  const formatDate = (s?: string | null) => {
    if (!s) return '—';
    const d = new Date(s);
    return isNaN(d.getTime()) ? '—' : d.toLocaleString('fr-FR');
  };

  // status badge moved to components/StatusBadge

  if (loading) {
    return (
      <div className="container mx-auto px-4 py-10">
        <div className="text-gray-600">Chargement des soumissions…</div>
      </div>
    );
  }
  if (error) {
    return (
      <div className="container mx-auto px-4 py-10">
        <div className="rounded border border-red-200 bg-red-50 p-4 text-red-700">{error}</div>
      </div>
    );
  }

  return (
    <div className="container mx-auto px-4 py-10 space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-2xl md:text-3xl font-bold">Vos Soumissions</h1>
        <Link href="/simulateur" className="text-sm underline">← Retour</Link>
      </div>

      {attempts.length === 0 ? (
        <div className="text-gray-600">Aucune soumission pour le moment.</div>
      ) : (
        <div className="space-y-4">
          {attempts.map(a => (
            <div key={a.id} className="border rounded-lg bg-white">
              <div className="p-4 flex items-center justify-between">
                <div className="space-y-1">
                  <div className="font-semibold">Tentative {a.id.slice(0,8)}</div>
                  <div className="text-xs text-gray-600">
                    Démarrée: {formatDate(a.started_at)} · Soumise: {formatDate(a.submitted_at)} · Durée: {Math.max(0, Math.floor((a.duration_seconds || 0) / 60))} min
                  </div>
                </div>
                <div className="flex items-center gap-3">
                  {statusBadge(a.status)}
                  <button
                    onClick={() => toggleOpen(a.id)}
                    className="text-sm border px-3 py-1.5 rounded hover:bg-gray-50"
                  >
                    {openId === a.id ? 'Masquer' : 'Voir la soumission'}
                  </button>
                  {openId === a.id && a.status === 'submitted' && (
                    <button
                      onClick={() => retrigger(a.id)}
                      className="text-sm border px-3 py-1.5 rounded hover:bg-gray-50"
                    >
                      Relancer l'évaluation
                    </button>
                  )}
                  {openId === a.id && (
                    <button
                      onClick={() => preview(a.id)}
                      className="text-sm border px-3 py-1.5 rounded hover:bg-gray-50"
                    >
                      Prévisualiser (sans sauvegarde)
                    </button>
                  )}
                  {openId === a.id && (
                    <button
                      onClick={() => runPerTask(a.id, true)}
                      className="text-sm border px-3 py-1.5 rounded hover:bg-gray-50"
                      disabled={Boolean(perTaskRunning[a.id])}
                    >
                      {perTaskRunning[a.id]?.dry ? `Prévisualiser par tâches (T${perTaskRunning[a.id].step || 0}/3…)` : 'Prévisualiser par tâches'}
                    </button>
                  )}
                  {openId === a.id && a.status === 'submitted' && (
                    <button
                      onClick={() => runPerTask(a.id, false)}
                      className="text-sm border px-3 py-1.5 rounded hover:bg-gray-50"
                      disabled={Boolean(perTaskRunning[a.id])}
                    >
                      {perTaskRunning[a.id] && !perTaskRunning[a.id]?.dry ? `Évaluer par tâches (éco) T${perTaskRunning[a.id].step || 0}/3…` : 'Évaluer par tâches (éco)'}
                    </button>
                  )}
                  <Link href={`/simulateur/attempts/${a.id}`} className="text-sm border px-3 py-1.5 rounded hover:bg-gray-50">Ouvrir</Link>
                </div>
              </div>

              {openId === a.id && (
                <div className="border-t p-4">
                  {detailsLoading[a.id] && (
                    <div className="text-gray-600 text-sm">Chargement des évaluations…</div>
                  )}
                  {!detailsLoading[a.id] && (!evals[a.id] || evals[a.id].length === 0) && (
                    <div className="text-gray-600 text-sm">
                      {['submitted','evaluating'].includes(attempts.find(x => x.id === a.id)?.status as any) ? 'Évaluation en cours… cela prend quelques secondes.' : 'Pas encore de feedback IA.'}
                      {['submitted','evaluating'].includes(attempts.find(x => x.id === a.id)?.status as any) && (
                        <div className="mt-3 rounded border p-3 bg-orange-50/50 text-xs text-orange-900">
                          <div className="font-semibold mb-1">Statut évaluation (système)</div>
                          <div className="flex flex-wrap gap-3 items-center">
                            <span className={`px-2 py-0.5 rounded ${health?.ok ? 'bg-emerald-100 text-emerald-700' : 'bg-red-100 text-red-700'}`}>{health?.ok ? 'OK' : (health?.loading ? 'Vérification…' : 'Problème')}</span>
                            <span>Modèle: <span className="font-medium">{health?.model || '—'}</span></span>
                            <span>Base URL: <span className="font-mono">{health?.baseUrl || '—'}</span></span>
                            <span>API Key: <span className={`font-medium ${health?.hasKey ? 'text-emerald-700' : 'text-red-700'}`}>{health?.hasKey ? 'présente' : 'absente'}</span></span>
                            <button
                              onClick={async () => {
                                try {
                                  setHealth({ loading: true });
                                  const h = await fetch('/api/simulator/health');
                                  const hj = await h.json();
                                  if (h.ok) {
                                    setHealth({ loading: false, ok: Boolean(hj?.ok), model: hj?.openrouter?.model, baseUrl: hj?.openrouter?.baseUrl, hasKey: Boolean(hj?.openrouter?.hasKey), timestamp: hj?.timestamp });
                                  } else {
                                    setHealth({ loading: false, ok: false, error: hj?.error || 'Health check failed' });
                                  }
                                } catch (e: any) {
                                  setHealth({ loading: false, ok: false, error: e?.message || 'Health check error' });
                                }
                              }}
                              className="text-orange-700 border border-orange-300 px-2 py-0.5 rounded hover:bg-orange-100"
                            >Rafraîchir</button>
                          </div>
                          {health?.timestamp && <div className="mt-1 text-[10px] text-gray-600">Dernier check: {new Date(health.timestamp).toLocaleTimeString('fr-FR')}</div>}
                          {evalErrors[a.id] && (
                            <div className="mt-2 text-[12px] rounded border border-red-200 bg-red-50 text-red-700 p-2">
                              Erreur évaluation: {evalErrors[a.id]}
                            </div>
                          )}
                          {health?.error && <div className="mt-1 text-[11px] text-red-700">{health.error}</div>}
                        </div>
                      )}
                    </div>
                  )}
                  {/* Show task statements and user's answers when no feedback yet */}
                  {!detailsLoading[a.id] && (!evals[a.id] || evals[a.id].length === 0) && detailsMap[a.id] && (
                    <div className="mt-4 space-y-4">
                      {detailsMap[a.id].tasks
                        .slice()
                        .sort((x, y) => (x.task_number ?? 99) - (y.task_number ?? 99))
                        .map((t) => {
                          const ans = (detailsMap[a.id].answers || []).find(u => u.task_number === t.task_number);
                          return (
                            <div key={t.task_number} className="rounded border bg-white p-3">
                              <div className="flex items-start justify-between">
                                <div className="font-semibold">Tâche {t.task_number}: {t.title}</div>
                                {(typeof ans?.word_count === 'number') && (
                                  <div className="text-xs text-gray-500">Mots utilisateur: {ans?.word_count}</div>
                                )}
                              </div>
                              {t.instructions && (
                                <div className="mt-2 text-xs text-gray-800"><span className="font-medium">Consignes:</span> {t.instructions}</div>
                              )}
                              {t.description && (
                                <div className="mt-1 text-xs text-gray-700 whitespace-pre-line">{t.description}</div>
                              )}
                              {Array.isArray(t.documents) && t.documents.length > 0 && (
                                <div className="mt-2 text-xs">
                                  <div className="font-medium mb-1">Documents</div>
                                  <ul className="list-disc list-inside space-y-1">
                                    {t.documents
                                      .slice()
                                      .sort((a: any, b: any) => (a.document_number ?? 99) - (b.document_number ?? 99))
                                      .map((d: any) => (
                                        <li key={d.id || d.document_number}>
                                          <span className="font-medium">D{d.document_number}.</span> {d.title}
                                          {d.content && (
                                            <div className="mt-1 text-[11px] text-gray-700 whitespace-pre-line border-l pl-2">
                                              {d.content}
                                            </div>
                                          )}
                                        </li>
                                      ))}
                                  </ul>
                                </div>
                              )}
                              <div className="mt-3 text-xs">
                                <div className="font-medium mb-1">Réponse de l'utilisateur</div>
                                {ans?.content ? (
                                  <div className="whitespace-pre-line border rounded p-2 bg-gray-50 text-gray-800">{ans.content}</div>
                                ) : (
                                  <div className="text-gray-500">Aucune réponse saisie pour cette tâche.</div>
                                )}
                              </div>
                            </div>
                          );
                        })}
                    </div>
                  )}
                  {/* Preview rendering when no saved evaluations yet */}
                  {!detailsLoading[a.id] && (!evals[a.id] || evals[a.id].length === 0) && Array.isArray(previewEvals[a.id]) && previewEvals[a.id].length > 0 && (
                    <div className="mt-4">
                      <div className="text-xs text-gray-600 mb-2">Aperçu local (non sauvegardé)</div>
                      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
                        {previewEvals[a.id]
                          .slice()
                          .sort((x, y) => (x.task_number ?? 99) - (y.task_number ?? 99))
                          .map((e, idx) => {
                            const currentTab = getTab(a.id, idx);
                            return (
                              <div key={idx} className="rounded border p-3 bg-white">
                                <div className="flex items-center justify-between mb-2">
                                  <div className="font-semibold">{e.task_number ? `T${e.task_number}` : 'Synthèse'}</div>
                                  <div className="flex gap-1 text-xs">
                                    <button className={`px-2 py-1 rounded border ${currentTab==='notes'?'bg-orange-100 border-orange-300':'hover:bg-gray-50'}`} onClick={()=>setTab(a.id, idx, 'notes')}>Notes</button>
                                    <button className={`px-2 py-1 rounded border ${currentTab==='feedback'?'bg-orange-100 border-orange-300':'hover:bg-gray-50'}`} onClick={()=>setTab(a.id, idx, 'feedback')}>Feedback</button>
                                    <button className={`px-2 py-1 rounded border ${currentTab==='checklist'?'bg-orange-100 border-orange-300':'hover:bg-gray-50'}`} onClick={()=>setTab(a.id, idx, 'checklist')}>Checklist</button>
                                    <button className={`px-2 py-1 rounded border ${currentTab==='c2'?'bg-orange-100 border-orange-300':'hover:bg-gray-50'}`} onClick={()=>setTab(a.id, idx, 'c2')}>Version C2</button>
                                  </div>
                                </div>

                                {currentTab === 'notes' && (
                                  <div className="text-sm space-y-1">
                                    {typeof e.score_20 === 'number' && (
                                      <div>Score: <span className="font-medium">{e.score_20}/20</span></div>
                                    )}
                                    {e.cecr_level && (
                                      <div>Niveau CECR: <span className="font-medium">{e.cecr_level}</span></div>
                                    )}
                                  </div>
                                )}

                                {currentTab === 'feedback' && (
                                  <div className="text-xs">
                                    {Array.isArray(e.positives) && e.positives.length > 0 && (
                                      <div className="mb-2">
                                        <div className="font-semibold mb-1">Points positifs</div>
                                        <ul className="list-disc list-inside space-y-0.5">
                                          {e.positives.map((p, i) => <li key={i}>{p}</li>)}
                                        </ul>
                                      </div>
                                    )}
                                    {Array.isArray(e.improvements) && e.improvements.length > 0 && (
                                      <div>
                                        <div className="font-semibold mb-1">Axes d'amélioration</div>
                                        <ul className="list-disc list-inside space-y-0.5">
                                          {e.improvements.map((p, i) => <li key={i}>{p}</li>)}
                                        </ul>
                                      </div>
                                    )}
                                    {!Array.isArray(e.positives) && !Array.isArray(e.improvements) && (
                                      <div className="text-gray-500">Aucun feedback détaillé.</div>
                                    )}
                                  </div>
                                )}

                                {currentTab === 'checklist' && (
                                  <div className="text-xs">
                                    {Array.isArray(e.improvements) && e.improvements.length > 0 ? (
                                      <ul className="list-disc list-inside space-y-0.5">
                                        {e.improvements.map((p, i) => <li key={i}>{p}</li>)}
                                      </ul>
                                    ) : (
                                      <div className="text-gray-500">Pas de checklist fournie.</div>
                                    )}
                                  </div>
                                )}

                                {currentTab === 'c2' && (
                                  <div className="text-xs whitespace-pre-line">
                                    {e.suggested_correction || <span className="text-gray-500">Version C2 non fournie.</span>}
                                  </div>
                                )}
                              </div>
                            );
                          })}
                      </div>
                    </div>
                  )}
                  {!detailsLoading[a.id] && Array.isArray(evals[a.id]) && evals[a.id].length > 0 && (
                    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
                      {evals[a.id]
                        .slice()
                        .sort((x, y) => (x.task_number ?? 99) - (y.task_number ?? 99))
                        .map((e, idx) => {
                          const currentTab = getTab(a.id, idx);
                          return (
                            <div key={idx} className="rounded border p-3 bg-white">
                              <div className="flex items-center justify-between mb-2">
                                <div className="font-semibold">{e.task_number ? `T${e.task_number}` : 'Synthèse'}</div>
                                <div className="flex gap-1 text-xs">
                                  <button className={`px-2 py-1 rounded border ${currentTab==='notes'?'bg-orange-100 border-orange-300':'hover:bg-gray-50'}`} onClick={()=>setTab(a.id, idx, 'notes')}>Notes</button>
                                  <button className={`px-2 py-1 rounded border ${currentTab==='feedback'?'bg-orange-100 border-orange-300':'hover:bg-gray-50'}`} onClick={()=>setTab(a.id, idx, 'feedback')}>Feedback</button>
                                  <button className={`px-2 py-1 rounded border ${currentTab==='checklist'?'bg-orange-100 border-orange-300':'hover:bg-gray-50'}`} onClick={()=>setTab(a.id, idx, 'checklist')}>Checklist</button>
                                  <button className={`px-2 py-1 rounded border ${currentTab==='c2'?'bg-orange-100 border-orange-300':'hover:bg-gray-50'}`} onClick={()=>setTab(a.id, idx, 'c2')}>Version C2</button>
                                </div>
                              </div>

                              {currentTab === 'notes' && (
                                <div className="text-sm space-y-1">
                                  {typeof e.score_20 === 'number' && (
                                    <div>Score: <span className="font-medium">{e.score_20}/20</span></div>
                                  )}
                                  {e.cecr_level && (
                                    <div>Niveau CECR: <span className="font-medium">{e.cecr_level}</span></div>
                                  )}
                                </div>
                              )}

                              {currentTab === 'feedback' && (
                                <div className="text-xs">
                                  {Array.isArray(e.positives) && e.positives.length > 0 && (
                                    <div className="mb-2">
                                      <div className="font-semibold mb-1">Points positifs</div>
                                      <ul className="list-disc list-inside space-y-0.5">
                                        {e.positives.map((p, i) => <li key={i}>{p}</li>)}
                                      </ul>
                                    </div>
                                  )}
                                  {Array.isArray(e.improvements) && e.improvements.length > 0 && (
                                    <div>
                                      <div className="font-semibold mb-1">Axes d'amélioration</div>
                                      <ul className="list-disc list-inside space-y-0.5">
                                        {e.improvements.map((p, i) => <li key={i}>{p}</li>)}
                                      </ul>
                                    </div>
                                  )}
                                  {!Array.isArray(e.positives) && !Array.isArray(e.improvements) && (
                                    <div className="text-gray-500">Aucun feedback détaillé.</div>
                                  )}
                                </div>
                              )}

                              {currentTab === 'checklist' && (
                                <div className="text-xs">
                                  {Array.isArray(e.improvements) && e.improvements.length > 0 ? (
                                    <ul className="list-disc list-inside space-y-0.5">
                                      {e.improvements.map((p, i) => <li key={i}>{p}</li>)}
                                    </ul>
                                  ) : (
                                    <div className="text-gray-500">Pas de checklist fournie.</div>
                                  )}
                                </div>
                              )}

                              {currentTab === 'c2' && (
                                <div className="text-xs whitespace-pre-line">
                                  {e.suggested_correction || <span className="text-gray-500">Version C2 non fournie.</span>}
                                </div>
                              )}
                            </div>
                          );
                        })}
                    </div>
                  )}
                </div>
              )}
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
